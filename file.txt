#include "push_swap.h"

static long	parse_and_validate(const char *str, int *i, int sign, t_stacks *s)
{
	long	result;

	if (!ft_isdigit(str[*i]))
		ft_error(s->a, s->b);
	result = 0;
	while (str[*i] >= '0' && str[*i] <= '9')
	{
		result = result * 10 + (str[(*i)++] - '0');
		if ((sign == 1 && result > INT_MAX) || (sign == -1
				&& result > (long)INT_MAX + 1))
			ft_error(s->a, s->b);
	}
	return (result);
}

int	ft_atoi(const char *str, t_stack *a, t_stack *b)
{
	long		result;
	int			sign;
	int			i;
	t_stacks	s;

	s.a = a;
	s.b = b;
	i = 0;
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] == '\0')
		ft_error(a, b);
	sign = 1;
	if (str[i] == '-' || str[i] == '+')
	{
		if (str[i] == '-')
			sign = -1;
		i++;
	}
	result = parse_and_validate(str, &i, sign, &s);
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] != '\0')
		ft_error(a, b);
	return ((int)(result * sign));
}
